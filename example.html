C++ Program to Multiply two Numbers
                
                    //6. C++ Program to Swap Two Numbers
                    #include 
                    using namespace std;
                    int main() {    
                        int num1,num2,temp;
                        cout    << "Enter first number say value in num1 : ";
                        cin     >> num1;
                        cout    << "Enter second number say value in num2 : ";
                        cin     >> num2;
                        temp    = num1;
                        num1    = num2;
                        num2    = temp;
                        cout    << "after swap \n";
                        cout    << "\n value in num1 = "  << num1 ;
                        cout    << "\n value in num2 = "  << num2 ;
                        return 0;
                    }
